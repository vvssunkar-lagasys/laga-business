-- Drop table if exists to ensure clean slate
drop table if exists public.customers;

-- Create Customers Table
create table public.customers (
    id uuid not null default gen_random_uuid(),
    created_at timestamp with time zone not null default now(),
    user_id uuid null,
    
    -- Company Details
    name text not null,
    gstin text null,
    email text null, -- Company email (if any)
    
    -- Primary Contact
    contact_name text null,
    contact_mobile text null,
    contact_phone text null,
    contact_email text null,
    pan_no text null,
    
    -- Billing Address
    billing_address1 text null,
    billing_address2 text null,
    billing_city text null,
    billing_pincode text null,
    state text null,        -- Billing State (mapped to 'state' in form)
    billing_country text null,

    -- Shipping Address
    shipping_title text null,
    shipping_address1 text null,
    shipping_address2 text null,
    shipping_city text null,
    shipping_pincode text null,
    shipping_state text null,
    shipping_country text null,
    
    -- Additional Contacts (JSONB Array)
    contacts jsonb default '[]'::jsonb,
    
    -- Constraints
    constraint customers_pkey primary key (id)
);

-- Enable RLS
alter table public.customers enable row level security;

-- Create Policy
create policy "Enable all access for authenticated users" on public.customers
    as permissive for all
    to authenticated
    using (true)
    with check (true);

    NEW.updated_at = NOW();
    RETURN NEW;
END;
